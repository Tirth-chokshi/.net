using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Web.UI;
using System.Web.UI.WebControls;

namespace WebApplication3
{
    public partial class WebForm1 : System.Web.UI.Page
    {
        protected void Page_Load(object sender, EventArgs e)
        {
            ScriptManager.ScriptResourceMapping.AddDefinition("jquery", new ScriptResourceDefinition
            {
                Path = "~/Scripts/jquery-3.6.0.min.js",  // Adjust path to your jQuery file
                DebugPath = "~/Scripts/jquery-3.6.0.js", // Adjust path to your jQuery debug file (optional)
                CdnPath = "https://ajax.googleapis.com/ajax/libs/jquery/3.6.0/jquery.min.js", // CDN URL
                CdnDebugPath = "https://ajax.googleapis.com/ajax/libs/jquery/3.6.0/jquery.js", // CDN debug URL (optional)
                CdnSupportsSecureConnection = true // Use HTTPS if the page is HTTPS
            });

        }

        protected void btnSubmit_Click(object sender, EventArgs e)
        {
            if (Page.IsValid)
            {
                lblSubmissionName.Text = "Name: " + Server.HtmlEncode(txtName.Text); //Important: Encode HTML to prevent XSS attacks.
                lblSubmissionEmail.Text = "Email: " + Server.HtmlEncode(txtEmail.Text);
                lblSubmissionPhone.Text = "Phone: " + Server.HtmlEncode(txtPhone.Text);
                lblSubmissionAddress.Text = "Address: " + Server.HtmlEncode(txtAddress.Text);
                lblSubmissionCity.Text = "City: " + Server.HtmlEncode(txtCity.Text);
                lblSubmissionState.Text = "State: " + Server.HtmlEncode(txtState.Text);
                lblSubmissionZipCode.Text = "Zip Code: " + Server.HtmlEncode(txtZipCode.Text);
                lblSubmissionCountry.Text = "Country: " + Server.HtmlEncode(txtCountry.Text);
                lblSubmissionComment.Text = "Comment: " + Server.HtmlEncode(txtComment.Text);
                lblSubmissionAge.Text = "Age: " + Server.HtmlEncode(txtAge.Text);

                // Make the submission details panel visible
                pnlSubmissionDetails.Visible = true;
             
            }
            else
            {
                // Validation failed.  The error messages will be displayed.
                // No need to do anything here; the validators handle the display.
                pnlSubmissionDetails.Visible = false; //Hide submission details panel if validation fails.
            }
        }
    }
}
